{# templates/i18n.html - macros for working with translations #} {# Basic translation macro #} {% macro t(key, fallback=None) %} {{ _(key) if fallback is none else _(key) or fallback }} {% endmacro %} {# Translation with pluralization #} {% macro n(singular, plural, n) %} {{ ngettext(singular, plural, n) }} {% endmacro %} {# Translation with parameters #} {% macro tr(key, **kwargs) %} {% set text = _(key) %} {{ text.format(**kwargs) if kwargs else text }} {% endmacro %} {# Context-aware translation for identical strings in different contexts #} {% macro ctx(context, key) %} {{ _(key + '|' + context) or _(key) }} {% endmacro %} {# Current language variable #} {% set current_lang = lang %}
